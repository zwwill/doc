H5MbInput Class

模块组件 - 移动端专用的国际手机号输入框[bH5MbInput]
基础的按钮封装组件
支持按钮上锁、loading、倒计时功能
使用方法：

【引用关键字】
require('@netease/urs-rui/src/module/bH5MbInput/bH5MbInput.js')
【组件标签】
<bH5MbInput  i18nMb={true} />
源码: src/module/bH5MbInput/bH5MbInput.js

  • 国际手机号

<div id="app" style="height: 300px;"></div>
</html>

<script>
var tpl =
    '<bH5MbInput ref=mb i18nMb={true}></bH5MbInput>'+
    '<button class="u-button" on-click={this.getValue()} >获取手机号</button>';
new URSRUI.Component({
    template: tpl,
    getValue:function () {
        alert(this.$refs.mb.getValue());

}).$inject('#app');

<script>
var tpl =
    '<bH5MbInput ref=mb i18nMb={true}></bH5MbInput>'+
    '<button class="u-button" on-click={this.getValue()} >getValue()</button>'+
    '<button class="u-button" on-click={this.setValue()} >setValue("13000000163")</button>'+
    '<button class="u-button" on-click={this.getFullValue()} >getFullValue()</button>'+
    '<button class="u-button" on-click={this.setFullValue()} >setFullValue("+1-|1234567|flag-US")</button>'+
    '<button class="u-button" on-click={this.bindMobile()} >bindMobile("13000000163")</button>'+
    '<button class="u-button" on-click={this.unBindMobile()} >bindMobile("")</button>';
new URSRUI.Component({
    template: tpl,
    getValue:function () {
        alert(this.$refs.mb.getValue());
    },
    setValue:function () {
        this.$refs.mb.setValue('13000000163');
    },
    getFullValue:function () {
        alert(this.$refs.mb.getFullValue());
    },
    setFullValue:function () {
        this.$refs.mb.setFullValue('+1-|1234567|flag-US');
    },
    bindMobile:function () {
        this.$refs.mb.bindMobile('13000000163');
    },
    unBindMobile:function () {
        this.$refs.mb.bindMobile('');
    }
}).$inject('#app');
</script>
