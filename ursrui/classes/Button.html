Button Class

单元组件 - 按钮[Button]
基础的按钮封装组件
支持按钮上锁、loading、倒计时功能
使用方法：

【引用关键字】
require('urs-rui/src/unit/button/button.js')
【组件标签】
<bButton clazz="bButton-sms f-fl" enableTip="获取验证码"></bButton>
源码: src/unit/bButton/bButton.js

<script>
var tpl =
    '<bButton ref=btn enableTip="点击锁定" on-countbegin={this.ll("计时开始",$event)} on-countend={this.ll("计时结束",$event)} count=3 on-click={this.lock()}></bButton>';
new URSRUI.Component({
    template: tpl,
    init:function () {

    },
    lock:function () {
        if(!this.ft){
            alert("锁定事件已经启动，请开启控制台【Ctrl+Shift+j(i)】或【option+command+i】观察效果");
            this.ft = true;
        }

        this.$refs.btn.setLock({count:1});
    },
    ll:function(info,event){
        console.log('info:'+info);
        console.log('event:');
        console.log(event);
    }
}).$inject('#app');

</script>
<html>
<div id="app"></div>
</html>

<script>
var tpl =
    '<bButton ref=btn enableTip="点击Loading" on-click={this.loading()}></bButton>';
new URSRUI.Component({
    template: tpl,
    init:function () {

    },
    loading:function () {
        this.$refs.btn.setLock({loading:1});
        setTimeout(function () {
            this.$refs.btn.unLock();
        }.bind(this),2000)
    }
}).$inject('#app');

</script>
