Pager Class

单元组件 - 翻页器[bPager]
基础的bPager封装组件
使用方法：

【引用关键字】
require('urs-rui/src/unit/bPager/bPager.js')
【组件标签】
<bPager clazz="m-page" ref=bPager total={total} limit={limit} currentPage={currentPage} on-page={this.onPage($event)}></bPager>
源码: src/unit/bPager/bPager.js

var tpl =
    '<bPager clazz="m-page" total={total} currentPage={currentPage} limit={limit} step={step} ref=bPager on-page={this.onPage($event)}></bPager>'
new URSRUI.Component({
    template: tpl,
    data:{
        total:100,
        limit:10,
        step:[2,3,2],
        currentPage:2
    },
    init:function(){
        setTimeout(function(){
            // this.$refs.bPager.setPage(9);
        }.bind(this),2000);
    },
    onPage:function(result){
        console.log(result.currentPage);
        this.data.currentPage = result.currentPage;
        this.$refs.bPager.updatePage(this.data);
    }
}).$inject('#app');
</script>
