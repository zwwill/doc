Tab Class

单元组件 - [bTab]
基础的bTab封装组件
使用方法：

【引用关键字】
require('urs-rui/src/unit/bTab/bTab.js')
【组件标签】
<bTab clazz="u-tab" ref=tab tabs={this.data.tabs}></bTab>

{Array Object}tabs：，
Object结构：{'name':'全部商品','key':'ALL'}
源码: src/unit/bTab/bTab.js

Demo

  • Tab
<html>
<div id="app"></div>
<style>
    #app .m-tab{overflow:hidden;}
    #app .u-tab{padding:0px 20px;border-left:solid 1px #ccc;}
    #app .bu-tab{height:40px;line-height:40px;margin-left:-1px;}
</style>
</html>

<script>
var tpl =
    '<bTab ref=bTab on-tab={this.onTab($event)} clazz="m-tab" tabs={this.data.tabs} />';
new URSRUI.Component({
    template: tpl,
    data:{
        tabs:[{key:1,name:"<p class='u-tab'>全部商品</p>"},{key:2,name:"<p class='u-tab'>审核（3）</p>"},{key:3,name:"<p class='u-tab'>已上架（1）</p>"}]
    },
    onTab:function(result){
        this.$refs.bTab.setTab(result.tabIndex)
    }
}).$inject('#app');
</script>
