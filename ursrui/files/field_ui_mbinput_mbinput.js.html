/**
 * 短信输入规则
 * @class MbInput
 * @type
 * @constructor
 * @param  {String} placeholder  placeholder文案，默认：'**手机号码**'；
 * @param  {Boolean} enableDelete  是否允许删除，默认：false；
 * @param  {Object} selected  默认选择：{nm:'CN',value:'86'}；
 * @param  {Boolean} open open；
 */


import './mbinput.scss'
import tpl from './mbinput.rgl'

var Validation = require('rgui-ui-field').Validation;
var Base = require('../input/input');
var ListView = require('../../../listview/listview');
var Item = require('../../../listview/item/item');
var Dom = require('regularjs').dom;

var MbInput = Base.extend({
    name: 'MbInput',
    template: tpl,
    /**
     * 初始化参数
     * @method config
     */
    config() {
        this.defaults({
            placeholder: '手机号码',
            enableDelete: false,
            selected: {nm:'CN',value:'86'},
            open: false
        });
        this.supr();  
    },
    /**
     * 初始化
     * @method init
     */
    init() {
         Dom.on(document, 'click', (e) => {
            var element = this.$refs.trigger;
            var element2 = e.target;
            while(element2) {
                if(element == element2)
                    return;
                element2 = element2.parentElement;
            }
            this.toggle(false);
        })
    },
    /**
     * select事件
     * @event on-select
     * */
    /**
     * onSelect事件方法
     * @method onSelect
     */
    onSelect (_data){
        this.data.selected = {nm:_data.selected.data.nm,value:_data.value};
        this.$update();
        this.toggle(false);
    },
    toggle (_open){
        this.data.open = _open;
        this.$update();
    }
});

module.exports = MbInput;
