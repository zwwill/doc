 * <bTab clazz="u-tab" ref=tab tabs={this.data.tabs}></bTab>
 * 
 * {Array Object}tabs：，
 * Object结构：{'name':'全部商品','key':'ALL'}
 *

 * @class Tab
 * @main Unit

 /**
 * tab切换结果
 * @event tab
 * @example <bTab on-tab={ function(_obj){console.log(_obj.tabIndex)} } />
 * @return {Object}
 */

'use strict'

import Regular from 'regularjs';
import tpl from './bTab.rgl';
import './bTab.scss';

let Tab = Regular.extend({
    template:tpl,
    name:'bTab',
    config:function(){
    },
    init:function(data){
        this.supr(data);
    },
    doTab:function(index){
        this.$emit('tab',{tabIndex:index});
        this.$update();
    },
    /**
     * 选择tab,高亮
     * @method setTab
     * @param {String} index 选择tab ** *ALL* |ALL1 | ALL2 **
     * @return undefined
     */
    setTab:function(index){
        this.data.tabIndexSelect = index;
    }
})

export default Tab;
