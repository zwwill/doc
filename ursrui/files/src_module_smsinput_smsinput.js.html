/**

 * @class SmsInput[待整理]

 * @show true

import './smsinput.scss'
import tpl from './smsinput.rgl'

var Validation = require('rgui-ui-field').Validation;
var Base = require('../../base/component');
var InputField = require('../../unit/base/input/input');

var SmsInputUI = Base.extend({
    name: 'SmsInput',
    template: tpl,
    config() {
        this.defaults({
            canSendTip: '获取验证码',
            retryTip: '重新获取',
            sendingTip: '[count]s后重发',
            status: 'canSend',
            count: 30,
            rule: [],
            disabled: false
        });
        this.supr();  
        this.watch();
    },
    watch() {
        this.$watch('status',(value)=>{
            this.data.disabled = (value === 'counting'||value === 'sending');
            switch(value){
                case 'counting':
                {
                    this.countDown();
                };
                default:{
                    this.$update();
                    return;
                }
            }
        })
    },
    init() {
        this.data.tip = this.data.canSendTip;
    },
    onSms (_event){
        this.$emit('getSms',{
            sender: this,
            status: this.data.status
        })
    },
    countDown(){
        var _count = this.data.count;
        var t = ()=>{
            setTimeout(()=>{
                this.data.tip = this.data.sendingTip.replace('[count]',_count);
                this.$update();
                _count --;
                
                if(_count <= 0){
                    this.data.status = 'retry';
                    this.data.tip = this.data.retryTip;
                    this.$update();
                    return;
                }
                t();
            },1000)
        }
        t();
    },
    handleEvent(_t,_e){
        this.$emit(_t,_e);
    }
});

module.exports = SmsInputUI;
