'use strict'


var Regular = require('regularjs');
var directive = require('./directive');
var filter = require('./filter');
var Dom = require('regularjs').dom;

var Component = Regular.extend({
    
    /**
     * @protected
     */
    config () {
        this.defaults({
            'clazz':'',
            visible: true
        });
        this.supr();
    },
    /**
     * @protected
     */
    reset () {
        this.data = {};
        this.config();
    },
    /**
     * 设置this.data的默认值，不会覆盖已经存在的key
     * 
     * @param {object} _args
     */
    defaults (_args) {
        for (var _key in _args) {
            if (_args.hasOwnProperty(_key) && this.data[_key]===undefined) {
                this.data[_key] = _args[_key];
            }
        }
    }
})
.directive(directive)
.filter(filter)
.event('enter', function(elem, fire){
  function update(ev){
    if(ev.which == 13){ // ENTER key
      ev.preventDefault();
      fire(ev); // if key is enter , we fire the event;
    }
  }
  Dom.on(elem, "keypress", update);
  return function destroy(){ // return a destroy function
    Dom.off(elem, "keypress", update);
  }
});

export default Component;
