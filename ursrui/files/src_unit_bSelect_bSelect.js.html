 /**
  * **单元组件 - 基础下拉选框[bSelect]**
  */

'use strict'

import Base from '../bListView/bListView'
import tpl from './bSelect.rgl'
import './bSelect.scss'
import {dom} from 'regularjs'

let bSelect = Base.extend({
    template:tpl,
    name:'bSelect',
    config(){
        this.defaults({
            show:false,
            selectData:undefined,
            type: 'click',
            placeholder: ''
        });
        this.supr();
    },
    init(data){
        this.supr(data);
        dom.on(document, 'click', (e) => {
            var element = this.$refs.selectbox;
            var element2 = e.target;
            while(element2) {
                if(element == element2)
                    return;
                element2 = element2.parentElement;
            }
            this.toggle(0);
        })
        // @NOTE 不做默认选中
    },
    onmouseOver(){
        if(this.data.type == 'hover'){
            this.toggle(1);     
        }
    },
    onmouseOut(){
        if(this.data.type == 'hover'){
            this.toggle(0);
        }       
    },
    onclick(){
        if(this.data.type == 'click'){
            this.toggle(1);
        }
    },
    selectInner(n){
        // 如果有传入列表项
        this.data.selectData.selected = this.data.selectData.items[n];
        this.$emit('select',this.data.selectData.selected);     
        this.toggle(0);       
    },
    select(item){
        this.supr(item);
        // 控制下拉菜单隐藏和显示
        this.toggle(0);         
    },
    toggle(flag){
        this.data.show = flag;
        this.$emit('toggle',flag);
        this.$update();
    }
})


export default bSelect;
