 /**
 * **单元组件 - alert弹框[bAlert]**<br>
 * 基础的bAlert封装组件<br>
 * **使用方法：**<br>

 * 【引用关键字】
 * import bAlert from 'urs-rui/src/unit/bAlert/bAlert.js';
 * ```
 * ```
 * 【组件标签】
 * bAlert.show({
            title:'提示', // 可传空，不显示tilte
            content:'<p style="color:red;">你好</p>',
            cancelBtn:'取消'
        }, this.ok.bind(this), 
        this.cancel.bind(this)
    )
 *
 * ```
 * @class Alert
 * @main Unit
 * @demo unit/bAlert/demo-1.html {alert}
 * @constructor
 * 
 * @show true
 * @author hzjiangchuanhua@corp.netease.com

 /*================组件可传参数==================*/

 /**
 * 点击关闭按钮回调函数
 * @property close
 * @default 默认关闭当前窗口
 * @type function
 */

import Base from '../../base/component'
import _tpl from './bAlert.rgl';
import './bAlert.scss';

const bAlert = Base.extend({
    name: 'bAlert',
    template: _tpl,
    config(){
        this.defaults({
            title:'提示',
            content: '',
            closeBtn: false,
            okBtn: true,
            cancelBtn: false,
        });
        this.supr();
    },
    init(){
        // 如果不是内嵌组件，则嵌入到document.body中
        if(this.$root === this)
            this.$inject(document.body);
    },
    /**
     *  关闭对话框
     *  @param {boolean} result 点击确认还是取消
     */
    close(result){
        this.$emit('close',{
            result: result
        });
        result? this.ok() : this.cancel();
    },
    // 点击确认按钮
    ok(){
        this.$emit('ok');
        this.destroy();
    },
    // 点击取消按钮
    cancel(){
        this.$emit('cancel');
        this.destroy();
    }
});


/**
 * 显示alert弹框
 * @method show
 * @param {object} data 
 * @param {function} ok 确认成功回调
 * @param {function} cancel 取消成功回调
 * @param {function} close 关闭成功回调
 * @return 弹窗实例对象
 * 
 */
bAlert.show = function(data, ok, cancel, close){
    var _data = data || {},
        close = close || function(){},
        ok = ok || function(){},
        cancel = cancel || function(){};
    var _modal = new this({data:_data});
    _modal.$on('close', close)
          .$on('ok', ok)
          .$on('cancel', cancel);
    
    return _modal;
}

// export default bAlert;
module.exports = bAlert;
